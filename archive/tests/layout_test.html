      runLayoutCheck() { const results = document.getElementById('results'); const output = document.getElementById('test-output'); results.style.display = 'block'; let html = ''; // Check sidebar position const sidebar = document.querySelector('.test-sidebar'); const sidebarRect = sidebar.getBoundingClientRect(); if (sidebarRect.left < 100) { html += '

      ✓ PASS: Sidebar is on left side

      '; } else { html += '

      ✗ FAIL: Sidebar is not on left side

      '; } // Check editor position const editor = document.querySelector('.test-editor'); const editorRect = editor.getBoundingClientRect(); if (editorRect.left > sidebarRect.right - 50) { html += '

      ✓ PASS: Editor is on right side

      '; } else { html += '

      ✗ FAIL: Editor is not on right side (might be in sidebar)

      '; } // Check status bar position const statusBar = document.querySelector('.test-status'); const statusRect = statusBar.getBoundingClientRect(); const editorContent = document.querySelector('.test-editor-content'); const contentRect = editorContent.getBoundingClientRect(); if (statusRect.top > contentRect.bottom - 10) { html += '

      ✓ PASS: Status bar is at bottom

      '; } else { html += '

      ✗ FAIL: Status bar is not at bottom

      '; } // Check widths if (Math.abs(sidebarRect.width - 280) < 10) { html += '

      ✓ PASS: Sidebar width is correct (~280px)

      '; } else { html += `

      ⚠ WARNING: Sidebar width is ${Math.round(sidebarRect.width)}px (should be ~280px)

      `; } if (editorRect.width > 500) { html += '

      ✓ PASS: Editor area is wide enough

      '; } else { html += `

      ✗ FAIL: Editor area is only ${Math.round(editorRect.width)}px wide (too small)

      `; } // Check overall layout const main = document.querySelector('.test-main'); const mainRect = main.getBoundingClientRect(); if (mainRect.height > 400) { html += '

      ✓ PASS: Main content has sufficient height

      '; } else { html += `

      ⚠ WARNING: Main content height is only ${Math.round(mainRect.height)}px

      `; } output.innerHTML = html; // Scroll to results results.scrollIntoView({ behavior: 'smooth' }); } // Run check on page load window.addEventListener('load', () => { setTimeout(runLayoutCheck, 1000); });
